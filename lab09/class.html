      scoreStudentBySid(sid) {
  const student = this.students.find(s => s.sid == sid);
  return student.answers
    .filter((answer) => this.question(answer).answer === answer.answer)
    .reduce((correct) => correct + 1, 0);
  }
getAverageScore() {
  return this.students.reduce((avg, student) => avg + this.scoreStudentBySid(student.sid), 0) / students.length;
}
