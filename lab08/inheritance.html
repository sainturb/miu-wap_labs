  3. Add a new method named sort without parameters in built-in constructor function Array. It'll sort all elements in the array in ascending order. 

    Array.prototype.mysort = function() {
  const sorted = this;
  const length = this.length;
  for(let i = length; i > 0 ; i--) {
    for(let j = 1; j < length; j++) {
      if (sorted[j - 1] > sorted[j]) {
        let temp = sorted[j-1];
        sorted[j-1]= sorted[j];
        sorted[j] = temp;
      }
    }
  }
  return sorted;
}
